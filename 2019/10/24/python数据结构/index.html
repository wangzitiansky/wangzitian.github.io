    List 列表

    list.append(x)
    list.append(num) # list[len(list):] = [num]
    list.insert(i, x)
    list.insert(0, num) # 插入列表头部
    list.remove(x)
    list.remove(num) # 删除第一个值为num的元素
    list.pop([i])
    list.pop(i) # 删除指定位置的元素

    列表推导式

    列表推导式提供了一个更简单的创建列表的方法

    # 创建一个平方列表
>>> list = [i * i for i in range(10)]
>>> list
[0, 1, 4, 9, 16, 25, 36, 49, 64, 81]

    列表表达式的构成

    # 列表推导式的结构是由一对方括号所包含的以下内容：一个表达式，后面跟一个 for 子句，然后是零个或多个 for 或 if 子句。 其结果将是一个新列表，由对表达式依据后面的 for 和 if 子句的内容进行求值计算而得出。
>>> [(x, y) for x in [1, 2, 3] for y in [1, 2, 3] if x != y]
[(1, 2), (1, 3), (2, 1), (2, 3), (3, 1), (3, 2)]
>>> [(x, y) for x in [1, 2, 3] for y in [1, 2, 3] if x == y]
[(1, 1), (2, 2), (3, 3)]

    将二维列表转成一维
    # 二维数组转一维
>>> list = [[1, 2, 3], [4, 5, 6], [7, 8, 9]]
>>> [i for row in list for i in row]
[1, 2, 3, 4, 5, 6, 7, 8, 9]

    set集合

    set对象是由具有唯一性的hashable对象组成的无序多项集合

    hashable

    一个对象在其生命周期中不可变，并且可以同其他对象比较则可哈希

    可哈希 —- 具有__hash__方法

    可比较 —– 具有__eq__方法

    大多数内置不可变对象可哈希

    可变容器都不可哈希

    如果不可变容器的内容可哈希 则不可变容器可哈希

    用户定义的类的实例对象可哈希 哈希值是id()

    heapq简介

    import heapq
# heapq实现了优先队列算法
# 是一个数组实现的二叉树 满足heap[k] <= heap[2*k+1] 和 heap[k] <= heap[2*k+2]
# 默认是最小堆
heap = []
item = ()
heapq.heappush(heap, item)
# item加入堆 并保证堆的不变性
heapq.heappop(heap)
# 弹出并返回堆顶 保持堆的不变性 堆为空则抛出IndexError
# 堆顶可以使用heapq[0]来访问
    排序字典
    from collections import OrderedDict
# OrderedDict 会保持着插入时的顺序 内部是一个双向链表

d = OrderedDict()
d['sky'] = 4
d['jkr'] = 3
d['little zz'] = -2
for key in d:
    print(key, d[key])

    字典运算

    用zip函数将字典变成(value, key)的元组序列
    prices = {
    'ACME': 45.23,
    'AAPL': 612.78,
    'IBM': 205.55,
    'HPQ': 37.20,
    'FB': 10.75
}
print(min(zip(prices.values(), prices.keys())))
print(max(zip(prices.values(), prices.keys())))
print(sorted(zip(prices.values(), prices.keys())))
